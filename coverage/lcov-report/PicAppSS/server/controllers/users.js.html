    create: function (req, res) {
 
        //se valida la ApiKey
        Server.find({
            where: {token: req.query.ApplicationToken}
        })
            .then(server => {
                if (!server) {
                    return res.status(401).send({
                        code: '401',
                        message: 'Unauthorized',
                    });
                }
                else {
 
                    //si el ApiKey se valido entonces se crea el usuario
 
                    User.create({
                        name: req.body.username,
                        password: req.body.password,
                        id: req.body.id,
                        ref: req.body._rev,
                        applicationOwner: req.body.applicationOwner,
                    })
                        .then(user => res.status(200).json({
                            metadata: {
                                version: "1.0"
                            },
                            user: {
                                id: user.id,
                                _rev: user.ref,
                                applicationOwner: user.applicationOwner,
                                username: user.name

                            res.status(400).send({
                                code: '400',

function hash(str){
 
    var text = "";
    var possible = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
 
    for (var i = 0; i < 5; i++)
        text += possible.charAt(Math.floor(Math.random() * possible.length));
 
    var hash = 0;
    if (text.length == 0) return hash;
    for (i = 0; i < text.length; i++) {
        char = text.charCodeAt(i);		hash = ((hash<<5)-hash)+char;		hash = hash & hash; // Convert to 32bit integer
    }
    if (hash < 0) {
        hash = hash * -1;
    }
    return hash;
