const Server = require('../models').Server;
const User = require('../models').User;
 
module.exports = {
    create: function (req, res) {
 
        //se valida la BusinessToken
        User.find({
            where: {token: req.query.BusinessToken}
        })
            .then(user => {
                if (!user) {
                    return res.status(401).send({
                        code: '401',
                        message: 'Unauthorized',
                    });
                }
                else {
 
                    Server.create({
                name: req.body.name,
                id: req.body.id,
                ref: req.body._rev,
                createdBy: req.body.createdBy,
                createdAt: req.body.createdTime,
                updatedAt: req.body.lastConnection,
                token: hash(String(req.body.name)),
 
            })
                    .then(server => res.status(201).json({
                        metadata: {
                            version: "1.0"
                        },
                        server: {
                            server: {
                                id: server.id,
                                _rev: server.ref,
                                createdBy: server.createdBy,
                                createdTime: server.createdAt,
                                name: server.name,
                                lastConnection: server.updatedAt
                            },
                            token: {
                                expiresAt: Date.now() + + 2*24*60*60*1000,
                                token: server.token
                            }
                        }
                    }))
                    .catch(error => {
                        res.status(400).send({code: '400',message:"Incumplimiento de precondiciones " +
                        "(parámetros faltantes) o validación fallida", error})
                    });
                }
            })
            .catch(error => res.status(500).send({code: '500', message:"Unexpected error", error}));
    },

function hash(str){
 
    var text = "";
    var possible = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
 
    for (var i = 0; i < 5; i++)
        text += possible.charAt(Math.floor(Math.random() * possible.length));
 
    var hash = 0;
    if (text.length == 0) return hash;
    for (i = 0; i < text.length; i++) {
        char = text.charCodeAt(i);		hash = ((hash<<5)-hash)+char;		hash = hash & hash; // Convert to 32bit integer
    }
    if (hash < 0) {
        hash = hash * -1;
    }
    return hash;
}
