    /* CREATE FILE*/
    create: function (req, res) {
 
        User.find({
            where: {token: req.query.BusinessToken}
        })
            .then(user => {
 
                if (!user) {
                    return res.status(401).send({
                        code:'401',
                        message: 'Unauthorized',
                    });
                }
 
                else {
                    File.create({
                        name: req.body.filename,
                        id: req.body.id,
                        rev: req.body._rev,
 
                        createdAt: req.body.createdTime,
                        updatedAt: req.body.updatedTime,
                        size: req.body.size,
                        resource: req.body.resource,
 
                    })
                        .then(file => res.status(201).json({
                            metadata: {
                                version: "1.0"
                            },
                            user: {
                                id: file.id,
                                _rev: file.rev,
                                createdTime: file.createdAt,
                                updatedTime: file.updatedAt,
                                size:file.size,
                                filename: file.name,
                                resource: file.resource
                            }

                            res.status(400).send({
                                code:'400',
                                message: "Incumplimiento de precondiciones " +

                            })
                        });
                }
 
            })
        .catch(error => res.status(500).send({
            code: '500',
